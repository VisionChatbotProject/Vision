CREATE TABLE quizs(quiz_id INTEGER PRIMARY KEY AUTOINCREMENT, quiz_name NVARCHAR(256) NOT NULL);
CREATE TABLE questions(question_id INTEGER PRIMARY KEY AUTOINCREMENT, quiz_id INTEGER, question_text NVARCHAR(256) NOT NULL,FOREIGN KEY (quiz_id) REFERENCES quizs (quiz_id) ON DELETE CASCADE);
CREATE TABLE answers(answer_id INTEGER PRIMARY KEY AUTOINCREMENT, question_id INTEGER, answer_text NVARCHAR(256) NOT NULL, is_correct BOOLEAN,FOREIGN KEY (question_id) REFERENCES questions (question_id) ON DELETE CASCADE);
